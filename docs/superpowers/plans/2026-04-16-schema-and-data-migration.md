## Task 1: Draft Postgres schema SQL

**Files:**
- Create: `server/internal/db/pg/schema.sql`

**Type translation rules** (SQLite → Postgres):
- `INTEGER PRIMARY KEY AUTOINCREMENT` → `BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY` (lets us insert explicit IDs during migration).
- `INTEGER` used as bool (`active`, `approved`, `enabled`, `is_primary`) → `BOOLEAN`. Conversion in migration: 0→false, non-zero→true.
- `INTEGER` for real integers (counts, fks, bytes) → `BIGINT`.
- `TIMESTAMP DEFAULT CURRENT_TIMESTAMP` → `TIMESTAMPTZ NOT NULL DEFAULT now()`. SQLite stores TIMESTAMP as naive UTC text; we parse as UTC.
- `TEXT` → `TEXT`.
- `INTEGER` used as epoch-ish (`user_sites.updated_at`) → `BIGINT` (keep as integer seconds; don't auto-convert).
- Foreign key clauses carry over verbatim (`ON DELETE CASCADE`, etc.).

```sql
-- Proxyness Postgres schema. Mirrors the SQLite schema in server/internal/db/db.go.
-- Booleans are real BOOL; timestamps are TIMESTAMPTZ; integer PKs use identity columns
-- so the migration tool can insert explicit IDs and keep FKs intact.

BEGIN;

CREATE TABLE IF NOT EXISTS users (
    id          BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name        TEXT NOT NULL,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS devices (
    id          BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id     BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name        TEXT NOT NULL,
    key         TEXT UNIQUE NOT NULL,
    active      BOOLEAN NOT NULL DEFAULT TRUE,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS traffic_stats (
    id          BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    device_id   BIGINT NOT NULL REFERENCES devices(id) ON DELETE CASCADE,
    hour        TIMESTAMPTZ NOT NULL,
    bytes_in    BIGINT NOT NULL DEFAULT 0,
    bytes_out   BIGINT NOT NULL DEFAULT 0,
    connections BIGINT NOT NULL DEFAULT 0
);
CREATE UNIQUE INDEX IF NOT EXISTS idx_traffic_device_hour ON traffic_stats(device_id, hour);

CREATE TABLE IF NOT EXISTS changelog (
    id          TEXT PRIMARY KEY,
    title       TEXT NOT NULL,
    description TEXT NOT NULL DEFAULT '',
    type        TEXT NOT NULL DEFAULT 'feature',
    created_at  TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS logs (
    id          BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    level       TEXT NOT NULL DEFAULT 'info',
    message     TEXT NOT NULL,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_logs_created_at ON logs(created_at);

CREATE TABLE IF NOT EXISTS sites (
    id                  BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    slug                TEXT UNIQUE NOT NULL,
    label               TEXT NOT NULL,
    primary_domain      TEXT UNIQUE NOT NULL,
    approved            BOOLEAN NOT NULL DEFAULT TRUE,
    created_by_user_id  BIGINT REFERENCES users(id) ON DELETE SET NULL,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE IF NOT EXISTS site_domains (
    site_id     BIGINT NOT NULL REFERENCES sites(id) ON DELETE CASCADE,
    domain      TEXT NOT NULL,
    is_primary  BOOLEAN NOT NULL DEFAULT FALSE,
    PRIMARY KEY (site_id, domain)
);
CREATE INDEX IF NOT EXISTS idx_site_domains_domain ON site_domains(domain);

CREATE TABLE IF NOT EXISTS site_ips (
    site_id  BIGINT NOT NULL REFERENCES sites(id) ON DELETE CASCADE,
    cidr     TEXT NOT NULL,
    PRIMARY KEY (site_id, cidr)
);

CREATE TABLE IF NOT EXISTS user_sites (
    user_id     BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    site_id     BIGINT NOT NULL REFERENCES sites(id) ON DELETE CASCADE,
    enabled     BOOLEAN NOT NULL DEFAULT TRUE,
    updated_at  BIGINT NOT NULL,
    PRIMARY KEY (user_id, site_id)
);

COMMIT;
```

```go
// server/cmd/migrate-pg/main.go
// Copies every table from a Proxyness SQLite DB into a Postgres DB with
// identical schema (see server/internal/db/pg/schema.sql). Preserves row IDs
// and advances identity sequences afterwards so newly-inserted rows don't collide.
package main

import (
	"context"
	"database/sql"
	"flag"
	"fmt"
	"log"
	"os"

	"github.com/jackc/pgx/v5"
	_ "modernc.org/sqlite"
)

func main() {
	sqlitePath := flag.String("sqlite", "", "path to source SQLite file (required)")
	pgURL := flag.String("pg", "", "destination Postgres connection URL (required)")
	skipLogs := flag.Bool("skip-logs", false, "skip the logs table (large, optional)")
	flag.Parse()

	if *sqlitePath == "" || *pgURL == "" {
		flag.Usage()
		os.Exit(2)
	}

	ctx := context.Background()

	src, err := sql.Open("sqlite", *sqlitePath)
	must(err, "open sqlite")
	defer src.Close()

	dst, err := pgx.Connect(ctx, *pgURL)
	must(err, "connect postgres")
	defer dst.Close(ctx)

	tx, err := dst.Begin(ctx)
	must(err, "begin tx")
	defer tx.Rollback(ctx) //nolint:errcheck // committed below on success

	// Order matters: parents before children.
	if err := copyUsers(ctx, src, tx); err != nil {
		log.Fatalf("users: %v", err)
	}
	if err := copyDevices(ctx, src, tx); err != nil {
		log.Fatalf("devices: %v", err)
	}
	if err := copyTrafficStats(ctx, src, tx); err != nil {
		log.Fatalf("traffic_stats: %v", err)
	}
	if err := copyChangelog(ctx, src, tx); err != nil {
		log.Fatalf("changelog: %v", err)
	}
	if !*skipLogs {
		if err := copyLogs(ctx, src, tx); err != nil {
			log.Fatalf("logs: %v", err)
		}
	}
	if err := copySites(ctx, src, tx); err != nil {
		log.Fatalf("sites: %v", err)
	}
	if err := copySiteDomains(ctx, src, tx); err != nil {
		log.Fatalf("site_domains: %v", err)
	}
	if err := copySiteIPs(ctx, src, tx); err != nil {
		log.Fatalf("site_ips: %v", err)
	}
	if err := copyUserSites(ctx, src, tx); err != nil {
		log.Fatalf("user_sites: %v", err)
	}
	if err := advanceSequences(ctx, tx); err != nil {
		log.Fatalf("advance sequences: %v", err)
	}

	if err := tx.Commit(ctx); err != nil {
		log.Fatalf("commit: %v", err)
	}
	fmt.Println("migration complete")
}

func must(err error, msg string) {
	if err != nil {
		log.Fatalf("%s: %v", msg, err)
	}
}
```

## Task 4: Implement per-table copy functions

**Files:**
- Create: `server/cmd/migrate-pg/copy.go`

**Pattern:** Each `copyX` reads all rows from SQLite with a typed `*sql.Rows` scan, then uses `pgx.CopyFrom` with a `CopyFromRows` source. `pgx.CopyFrom` is the fast path — avoids individual INSERTs. Booleans converted from SQLite int. Nullable FKs (`created_by_user_id`) handled with `sql.NullInt64` → `pgtype.Int8{Valid:bool}`. For `TIMESTAMP` columns stored as text in SQLite (format `2006-01-02 15:04:05`), parse as UTC.

- [ ] **Step 1: Write copy.go with all 9 copy functions**

```go
package main

import (
	"context"
	"database/sql"
	"fmt"
	"time"

	"github.com/jackc/pgx/v5"
)

const sqliteTimeLayout = "2006-01-02 15:04:05"

func parseSQLiteTime(s string) (time.Time, error) {
	if t, err := time.Parse(time.RFC3339, s); err == nil {
		return t.UTC(), nil
	}
	return time.ParseInLocation(sqliteTimeLayout, s, time.UTC)
}

func copyUsers(ctx context.Context, src *sql.DB, tx pgx.Tx) error {
	rows, err := src.QueryContext(ctx, `SELECT id, name, created_at FROM users ORDER BY id`)
	if err != nil {
		return err
	}
	defer rows.Close()
	var data [][]any
	for rows.Next() {
		var id int64
		var name, createdAt string
		if err := rows.Scan(&id, &name, &createdAt); err != nil {
			return err
		}
		ts, err := parseSQLiteTime(createdAt)
		if err != nil {
			return fmt.Errorf("users.created_at %q: %w", createdAt, err)
		}
		data = append(data, []any{id, name, ts})
	}
	if err := rows.Err(); err != nil {
		return err
	}
	n, err := tx.CopyFrom(ctx, pgx.Identifier{"users"}, []string{"id", "name", "created_at"}, pgx.CopyFromRows(data))
	if err != nil {
		return err
	}
	fmt.Printf("users: copied %d\n", n)
	return nil
}

func advanceSequences(ctx context.Context, tx pgx.Tx) error {
	// Identity columns silently use an internal sequence. After inserting explicit IDs,
	// the sequence's last_value is still 1 — next INSERT without explicit id would collide.
	// setval() to max(id) on each table that has an identity column.
	tables := []string{"users", "devices", "traffic_stats", "logs", "sites"}
	for _, t := range tables {
		sql := fmt.Sprintf(`SELECT setval(pg_get_serial_sequence('%s','id'), COALESCE((SELECT MAX(id) FROM %s), 1), (SELECT COUNT(*) FROM %s) > 0)`, t, t, t)
		if _, err := tx.Exec(ctx, sql); err != nil {
			return fmt.Errorf("setval %s: %w", t, err)
		}
	}
	return nil
}
```
